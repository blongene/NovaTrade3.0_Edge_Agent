# PowerShell
Get-Content .env | ForEach-Object {
  if ($_ -match '^\s*#') { return }
  if ($_ -match '^\s*$') { return }
  $k,$v = $_ -split '=',2
  [Environment]::SetEnvironmentVariable($k,$v,'Process')
}
